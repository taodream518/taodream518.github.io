    Git学习笔记

    CoderTao 2020/2/1 Git

    # 一、git常用命令

    git init									初始化
git add										添加文件
git status									查看状态
git commit									提交
git commit --amend							修改最新一次提交的备注
git push									推送到仓库
git pull									从远程仓库拉取数据
git clone									从远程仓库拷贝数据
git remote add origin http://xxx.com 		关联远程仓库
git remote set-url origin http://xxx.com 	重新关联远程仓库
git reset --sort HEAD^

    # 二、git配置相关

    git config --global user.name 'tao'				配置全局用户名
git config --global user.email '10086@qq.com'  	配置全局用户邮箱
git config --global alias.c commit				配置commit别名

    # 三、branch常用命令

    git branch 			查看当前所在分支
git branch -a 		查看所有分支
git branch dev 		创建本地dev分支
git checkout -b dev 创建本地dev分支并切换到该分支
git branch -d dev 	删除本地dev分支
git push origin -d dev 删除远程dev分支

    # 四、tag常用命令

    # 打标签
git tag v1.0.0 -m'注释'	 	 # 打标签

# 查看标签
git tag --list 				# 查看本地所有标签
git ls-remote --tag			# 查看远程所有标签
git show v1.0.0 			# 查看标签的提交信息

# 推送标签
git push origin v1.0.0 		# 推送特定标签到远程
git push origin --tags  	# 推送本地所有标签到远程

# 删除标签
git tag -d v1.0.0		    # 删除本地特定标签
git push origin -d v1.0.0   # 删除远程特定标签

    # 五、rebase常用命令

    git rebase -i HEAD~4 		# 最近4次的commit，最近一次提交在最下面
=====================
git add .
git rebase --continue 	# 如需继续，需执行以上2条命令 （可能会出现冲突情况，需解决）
git rebase --abort 			# 如需放弃，需执行此命令
git push -f origin test # 需强制推送

    pick： 保留该 commit
    reword： 保留该 commit，但我需要修改该 commit 的注释
    edit： 保留该 commit, 但我要停下来修改该提交(不仅仅修改注释)
    squash： 将该 commit 和前一个 commit 合并，时间上的前一个
    fixup： 将该 commit 和前一个 commit 合并，但我不要保留该提交的注释信息
    exec： 执行 shell 命令
    drop： 丢弃该 commit

    # 一些额外记录
git add index.html 				添加index.html
git rm --cached index.html		移除index.html
git reset HEAD index.html 		第二次或以上从暂存区移除
git add *.html					添加html的所有文件
git add .						添加所有文件
git commit 						提交后会进入修改备注页面，按esc键退出即可
git commit -m '备注的信息'   	  提交同时备注
git commit --amend				修改最新一次提交的备注
git branch 						查看分支状态
git branch login				创建分支login
git checkout login				切换到login分支下
git checkout -b fz1				创建fz1分支并切换到该分支下
git checkout master				切换到主线
git checkout -- index.html 		把index.html文件撤回到被修改前的版本
git merge login					把分支合并到主线上(必须是在主线上合并)
git branch --merged 			查看已合并的分支
git branch --no-merged 			查看尚未合并的分支
git branch -d test				删除已经合并的分支
git branch -D test				删除尚未合并的分支
git remote
git remote add origin https://github.com/taodream518/homework.git
git push
git clone https://github.com/taodream518/homework.git
git mv a.html b.html			文件重命名
git log 						查看日志
git log -p 						显示文件变动
git log -1 						查看日志最后一次的变动信息
git log -2 						查看日志倒数第二次的变动信息
git log --oneline				经典显示
git log --name-only				查看哪些文件发生变化
git log --name-status			查看哪些文件发生了什么类型的变化
git stash 						存到暂存区(在文件还没写完,但需要切换到别的分支，必须在add或commit之后操作)
git stash list	    			查看暂存区的文件
git stash apply					恢复暂存区
git stash drop stash@{0}		删除暂存区
git stash pop 					恢复暂存区中的文件并删除
git tag v1.0					标签
git archive master --prefix='dist/' --forma=zip > dist.zip	项目打包成压缩包
git rebase master				HEAD指针向后移
cd.>.gitignore					创建一个忽略某些文件的文件
cd.>README.md					创建一个

    3、Github 使用流程：

    1. 在 Github 官网创建账号
    2. 创建远程仓库(用来存储代码)
    3. 安装 Github 客户端
    4. 登录客户端并克隆仓库
    5. 在本地仓库中存储数据
    6. 提交数据并备注信息
    7. 同步本地数据到远程仓库
