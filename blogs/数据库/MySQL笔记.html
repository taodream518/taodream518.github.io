    CoderTao 2024/1/23 MySQL数据库

    # 一、 数据库介绍

    # 二、 SQL 结构化查询语言

    # 语言类型

    # 1. DDL (Data Definition Language) 数据定义语言

    用于对数据库对象（ 数据库、数据表、索引、视图 ）的创建、修改、删除。

    • CREATE 创建新的数据库、表、索引、视图等。
    • ALTER 修改已存在的数据库结构或表结构。
    • DROP 删除数据库对象如表、索引、视图等。
    • TRUNCATE 清空表中的所有数据，但保留表结构。
    • RENAME 重命名数据库对象。
    • COMMENT 为数据库对象添加注释。
    # 数据库操作

    查询数据库

    # 查询当前数据库列表
SHOW DATABASES;

# 查询数据库的创建 SQL 语句
SHOW CREATE DATABASE database_name;

    创建数据库

    # 创建数据库并同时指定字符集
CREATE DATABASE [IF NOT EXISTS] database_name CHARACTER SET utf8;

    修改数据库

    # 修改数据库的字符集
ALTER DATABASE database_name CHARACTER SET gbk;

    删除数据库

    # 删除数据库
DROP DATABASE [IF EXISTS] database_name;
    # 数据表操作

    创建数据表

    -- # 语法
CREATE TABLE [IF NOT EXISTS] table_name (
    column_name1 datatype1,
    column_name2 datatype2
    -- 定义主键
    [CONSTRAINT pk_name] PRIMARY KEY (column_name) COMMENT '主键',
    -- 定义外键以及设置级联操作 
    -- 级联操作 CASCADE：连带删除子表中所有与之相关的记录。 SET NULL：将子表中外键字段设为NULL（前提是该字段允许NULL值）。 RESTRICT：如果子表中有相关记录，则不允许删除父表中的记录（默认）。NO ACTION：具体取决于数据库系统。
    [CONSTRAINT fk_name] FOREIGN KEY (column_name) REFERENCES table_name (column_name) 
    ON DELETE CASCADE ON UPDATE CASCADE
)


-- 例如 创建文章表
CREATE TABLE IF NOT EXISTS article (
    id INT AUTO_INCREMENT COMMENT '文章ID',
    title VARCHAR(100) NOT NULL UNIQUE COMMENT '文章标题',
    content TEXT COMMENT '文章内容',
    author_id INT NOT NULL COMMENT '作者ID',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    CONSTRAINT pk_id PRIMARY KEY (id) COMMENT '主键',
    CONSTRAINT fk_author FOREIGN KEY (author_id) REFERENCES tb_users (id)
    ON DELETE CASCADE
    ON UPDATE CASCADE
);

    查看数据表结构

    DESC article;

    删除数据表

    DROP TABLE [IF EXISTS] table_name;

    修改数据表

    # 修改表名
-- # 语法
ALTER TABLE table_name RENAME TO new_table_name;
-- 例如
ALTER TABLE article RENAME TO post;

# 添加字段
-- # 语法
ALTER TABLE table_name ADD column_name datatype;
-- 例如
ALTER TABLE article ADD status TINYINT(1);


# 修改字段名和类型
-- # 语法
ALTER TABLE table_name CHANGE old_column_name new_column_name datatype;
-- 例如
ALTER TABLE article CHANGE title title1 VARCHAR(50);

    # 2. DML (Data Manipulation Language) 数据操作语言

    用于对数据库中的数据进行增删改查。

    • INSERT 向表中插入新的记录。
    • UPDATE 更新表中已存在的记录。
    • DELETE 从表中删除记录。

    插入数据

    -- # 语法
INSERT INTO table_name (column_name1, column_name2, ...) VALUES (value1, value2, ...);

-- 例如
INSERT INTO article (title, content, author_id) VALUES ('标题', '内容', 1);

    修改数据

    -- # 语法
UPDATE table_name SET column_name = value WHERE condition;

-- 例如
UPDATE article SET title = '标题2' WHERE id = 1;

    删除数据

    -- # 语法
DELETE FROM table_name WHERE condition;

-- 例如
DELETE FROM article WHERE id = 1;

    # 3. DQL (Data Query Language) 数据查询语言

    用于从数据库中查询和检索数据。

    # 4. DCL (Data Control Language) 数据控制语言

    用于完成事务管理等控制性操作。

    # 数据类型

    • 数值类型
    • 日期类型
    • 字符串类型

    # 字段约束

    • NOT NULL 非空约束，限制字段不能为空
    • UNIQUE 唯一约束，限制字段的值唯一
    • AUTO_INCREMENT 自增约束，用于自增字段的值
    • DEFAULT 默认值约束，当字段为空时，会自动设置为默认值
    • PRIMARY KEY 主键约束，限制字段唯一，不能重复，用于唯一标识表中的记录
    • FOREIGN KEY 外键约束，限制字段的值只能是另一个表中的值（ 建立表与表之间的关联关系 ）

    # 三、连接查询

    # 1. 内连接 (INNER JOIN)

    内连接：仅返回两个表中具有匹配值的行。

    注意： 连接的条件是 ON 之后的条件，不是 WHERE 之后的条件。WHERE 是查询两个表的 笛卡尔积 后再通过 WHERE 来限制的，性能差；而 ON 是在查询两个表之前就限制的，性能好。

    -- #语法
SELECT column_name(s)
FROM table1
INNER JOIN table2
ON table1.column_name = table2.column_name;


-- 例如
SELECT a.title, a.user_id, u.id, u.username
FROM tb_articles AS a
INNER JOIN tb_users AS u
ON u.id = a.user_id;

    # 2. 左连接 (LEFT JOIN)

    左连接：左连接会返回左表的所有记录，以及与之匹配的右表中的记录。如果右表中没有匹配项，则对应字段显示为 NULL。

    -- #语法
SELECT column_name(s)
FROM table1
LEFT JOIN table2
ON table1.column_name = table2.column_name;


-- 例如
SELECT a.title, a.user_id, u.id, u.username
FROM tb_articles AS a
LEFT JOIN tb_users AS u
ON u.id = a.user_id;

    # 3. 右连接 (RIGHT JOIN)

    右连接：返回右表的所有记录和左表中与之匹配的记录，如果没有匹配，则左侧显示为 NULL。

    -- #语法
SELECT column_name(s)
FROM table1
RIGHT JOIN table2
ON table1.column_name = table2.column_name;


-- 例如
SELECT a.title, a.user_id, u.id, u.username
FROM tb_articles AS a
RIGHT JOIN tb_users AS u
ON u.id = a.user_id;

    # 4. 子查询 (SUB QUERY)

    子查询：是在一个 SQL 查询语句内部嵌套的另一个 SQL 查询，通常用作表、列或表达式的值。子查询先执行，然后将结果传递给外部查询。

    -- #语法（ 单查询 ）
SELECT column_name(s)
FROM table1
WHERE column_name = (
    SELECT column_name FROM table2 WHERE = xxx
);


-- 例如
SELECT id, title, content
FROM tb_articles
WHERE user_id = (
	SELECT id FROM tb_users WHERE id = 1
);


    -- #语法（ 多查询 ）
SELECT column_name(s)
FROM table1
WHERE column_name IN (
    SELECT column_name FROM table2 WHERE (NOT) IN (value1, value2, ...)
);


-- 例如 查询用户ID 1 和 3 的文章
SELECT id, title, content
FROM tb_articles
WHERE user_id IN (
	SELECT id FROM tb_users WHERE id IN (1, 3)
);

-- 或者用虚拟表

-- 定义临时用户表
WITH temp_users AS (
    SELECT id
    FROM tb_users
    WHERE id IN (1, 3)
)
SELECT a.id, a.title, a.content
FROM tb_articles AS a
JOIN temp_users AS tu ON a.user_id = tu.id;

    # 四、存储过程（Stored Procedure）

    存储过程：是一种可以重复使用的 SQL 代码块，可以包含 SQL 语句、变量、函数、条件和循环等。可以理解为一个函数，但是可以包含 SQL 语句。

    # 1. 存储过程的创建和调用

    
-- #语法: 创建或替换存储过程
CREATE OR REPLACE PROCEDURE procedure_name(IN p1 INT, IN p2 INT)
BEGIN
    -- SQL 语句
END;
-- #语法: 调用存储过程
CALL procedure_name(p1, p2);


-- 例如
DROP PROCEDURE IF EXISTS GetArticlesByUserIds;
CREATE PROCEDURE GetArticlesByUserIds(IN uid1 INT, IN uid2 INT)
BEGIN
  SELECT id, title, content
  FROM tb_articles
  WHERE user_id IN (uid1, uid2);
END;

-- 调用存储过程
CALL GetArticlesByUserIds(1, 3);
